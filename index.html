// Align service card buttons and balance content document.addEventListener("DOMContentLoaded", function () { const cards = document.querySelectorAll(".bg-white.rounded-2xl.shadow-lg"); cards.forEach(function (card) { // bottom section: the one with the list + button const bottom = card.querySelector(".p-8:last-of-type"); if (!bottom) return; bottom.style.display = "flex"; bottom.style.flexDirection = "column"; bottom.style.height = "100%"; // push the button down with a softer gap const btn = bottom.querySelector("button.w-full"); if (btn) { btn.style.marginTop = "1rem"; } }); });
